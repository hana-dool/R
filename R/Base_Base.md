``` r
# transform
# 새로운 열을 추가해준다.
df <- data.frame(이름= c('서연','서윤','민준','하윤','현우'),
                 반=c('1반','1반','2반','2반','2반'), 
                 score = c(55,95,70,85,50))
# transform 을 통해서 새로운 열을 만드는 작업을 한번에 할 수 있다.
transform(df,new_score = score + 20) # new_score 라는 새로운 열을 추가
```
